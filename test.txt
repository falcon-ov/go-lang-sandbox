//импорты и пакет
func main() {
	var wg sync.WaitGroup
    var mu sync.Mutex
    counter := 0

	for i := 0; i < 100; i++ {
		wg.Add(1)
		go func() {
        mu.Lock()
			counter++ //ошибка data race много корутин пытаются обратиться к одной перемнной, фикс добавить mutex
        mu.Unlock()
			wg.Done()
		}()
	}

	wg.Wait()
	fmt.Println("Counter:", counter)
}
